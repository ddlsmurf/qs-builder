  def write_to_output filename, content = nil # :yields: file
    output_path = @output_root + filename
    @logger.info "Writing #{content ? content.length.to_s + "b " : "output"} to #{output_path}#{dry_run ? " (skipped - dry run)" : ""}" do
      unless dry_run
        folder = output_path.dirname
        folder.mkpath unless folder.directory?
        File.open(output_path, 'w') do |f|
          if content
            f.write(content)
          else
            yield f
          end
        end
      end
    end
  end
