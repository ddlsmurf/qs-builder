  def column id = nil, options = {}
    unless id
      id = options[:id]
      raise "Need a column ID" if id.nil?
    end
    if options.empty?
      @cols << {}
    else
      @cols << options
    end
    @col_count = @cols.size
    @col_ids << id
  end
