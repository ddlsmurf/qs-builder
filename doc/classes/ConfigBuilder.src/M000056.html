  def add_to_option_parser parser, default_group_name = "General Options"
    keys = @opts.keys.sort
    keys << keys.shift() if keys.first == ""
    keys.each do |k|
      label = k || default_group_name
      opts.separator ""
      opts.separator "#{label}:" if label
      @opts[k].each { |o| parser.send(:on, *o[0], &o[1]) }
    end
  end
