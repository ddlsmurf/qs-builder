    def each_handler name, &block # :yields: extension
      return enum_for(:each_handler, name) unless block_given?
      point = extension_point(name)
      ext = point[:handlers]
      done = []
      while ext && !ext.empty?
        ext.each(&block)
        done += ext
        ext = point[:handlers] - done
      end
      nil
    end
