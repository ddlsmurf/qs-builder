  def get_qs_segments obj, type
    values = []
    unless obj.is_a?(QS::Plugin) || obj.is_a?(Bundle)
      values += get_qs_segments(obj.plugin, nil)
    end
    values << obj.class.name.gsub("QS::", "")
    if type.nil? && (obj.is_a?(QS::Plugin) || obj.is_a?(Bundle))
      values << obj.name
    else
      values << obj.id
    end
    values << type if type
    values
  end
