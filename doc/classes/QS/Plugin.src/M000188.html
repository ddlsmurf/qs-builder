    def each &blk
      return enum_for(:each) unless blk
      each_catalog_preset.map{|l,d|l}.each(&blk)
      items = objects_to_index.flatten.select { |e| !e.is_a?(CatalogPreset) }
      until items.empty?
        stack = []
        items.each do |o|
          yield o
          stack += Array(o.objects_to_index) if o.respond_to? :objects_to_index
        end
        items = stack.flatten
      end
    end
