    def initialize info, id, parent
      if info.is_a?(Array)
        info = {:files => info}
      elsif info.is_a?(String)
        if info =~ /^\[([^\]]+)\]:(.+)$/
          info = {"bundle" => $1, "resource" => $2}
        else
          info = { :files => [info] }
        end
      end  
      super info, id, parent
    end
