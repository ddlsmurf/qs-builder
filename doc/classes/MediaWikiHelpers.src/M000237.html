  def html_links_to_wikimedia str
    return str unless str.is_a?(String)
    str.gsub(/<\s*a\b([^>]+)>/i) do |match|
      attrs = {}
      $1.scan(/([^ =]+)=(?:'([^']+)'|"([^"]+)"|([^ >]+))/i) do |matches|
        attrs[$1] = matches.drop(2).select {|e| e}.first
      end
      raise "Failed to understand html #{str.inspect}" unless attrs['href']
      "[#{attrs['href']} "
    end.gsub(/<\/a>/i, "]")
  end
