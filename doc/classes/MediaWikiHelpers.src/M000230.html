  def template_call name, *args
    args += args.pop.map { |k, v| "#{k} = #{v}" if v && v.to_s.length > 0 } if args.last.is_a?(Hash)
    args = args.map { |e| e.to_s }
    args.pop while args.last == ""
    args = args.join("\n| ")
    "{{#{name}#{args.length == 0 ? "" : "| #{args}"}}}"
  end
